## Hotel Flow Code


```javascript
export class HotelFlow extends Flow {
  public constructor() {
    super(HotelFlow);
  }

  protected defineSteps(): Step[] {
    return [
      new ExploreStep(this, true).setTemperature(0.5),
      new PresentStep(this, false),
      new CompareStep(this, false),
      new EndStep(this).useMemory('end'),
    ];
  }
}
```
